<distinct> WDJ DAU: 从 common 算Wandoujia 整体的 DAU 
select count(distinct id) from muce_topic.common where p_product = 'ANDROID' and p_date='20150401'
<distinct> WDJ NDLU: 从 applicationstart 算Wandoujia 整体的 NDLU 
select count(distinct id) from muce_topic.applicationstart where is_start_new = 1 and p_product='ANDROID' and p_date='20150401'
<group by> 分 bd_source_alias 来看 ANDROID 点击用户数 
select count(distinct id),bd_source_alias from muce_topic.click WHERE p_product='ANDROID' and p_date>='20150401' and p_date<='20150403' group by bd_source_alias
<like> <in> Android Music Distribution: 来自推送和播放消费来源的 Music 的下载与在线播放次数 
select count(*) from muce_topic.consumption where consumption_type in ('DOWNLOAD','ONLINE_PLAY') and content_type like '%MUSIC%' and consumption_source in ( 'push','player') and p_product='ANDROID' and p_date='20150401'
<limit> 查询 taskevent 的前10条日志记录 
select * from muce_event_client.taskevent where p_date='20150401' and p_hourmin=1200 limit 10
